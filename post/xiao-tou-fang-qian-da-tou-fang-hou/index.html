    // 小于头部元素放前面，大于头部元素放后面
void fnMoveElementsBetweenHead(int target[], int length) {
    int temp;
    int i = 0;
    int j = length-1;
    temp = target[i];
    while(i<j) {
        while(i<j && target[j]>temp) --j;
        // 从后往前，在第一个比temp小的元素停下来
        if (i<j) {
            target[i] = target[j];
            ++i;  // i后移
        }

        while(i<j && target[i]<temp) ++i;
        // 从前往后，在第一个比temp大的元素停下来
        if (i<j) {
            target[j] = target[i];
            --j;
        }
    }
    target[i] = temp;
}
