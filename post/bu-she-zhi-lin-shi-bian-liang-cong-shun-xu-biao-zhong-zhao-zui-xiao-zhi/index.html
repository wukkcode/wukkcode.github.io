    • 不设置临时变量从顺序表中找最小值

    挺有趣的一道题，记录下来了...

    int fnFindMinValue(int target[], int length) {
    int i = target[0];  // i是已知的
    while (i / 10 < length) {
        if (i % 10 > target[i / 10]) {
            i = i - i % 10;
            i = i + target[i / 10];
        }
        i = i + 10;
    }
    return i % 10;
}
